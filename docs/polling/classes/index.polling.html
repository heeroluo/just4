Class Polling

轮询类。

example
import { Polling } from '@just4/polling';
const polling = new Polling(() => {
  return new Promise((resolve) => {
    setTimeout(() => {
      console.log('executed');
      resolve();
    }, 1000);
  });
}, {
  interval: 2000
});
polling.start();
